# M8: Local Timeline Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement a fully local timeline feature — no OAuth, no external calendar sync — allowing users to create, edit, search, and view events with optional links to existing recordings and workspace documents.

**Architecture:** The Rust backend exposes a `timeline::manager::TimelineManager` that wraps all SQLite CRUD for `timeline_events`; tags are stored as a JSON string in SQLite and transparently serialized to/from `Vec<String>` in Rust. Seven Tauri commands (via `commands/timeline.rs`) bridge the manager to the frontend. The React side uses a single Zustand store (`store/timeline.ts`) and four components: `TimelinePanel` (navigation + mini-calendar + search), `TimelineMain` (month/week/day grid views), `EventModal` (create/edit form), and `EventCard` (compact event tile).

**Tech Stack:** Rust · sqlx 0.8 · serde_json · uuid · date-fns · Zustand · React 18 · TypeScript · Tailwind CSS · shadcn/ui

**Prerequisites:** M1 (scaffold, AppState, AppError), M2 (SQLite pool + migrations, `timeline_events` table already created in `0001_initial.sql`). M6 (Workspace) should be complete so that `workspace_documents` rows exist for `link_event_to_document` to reference.

  ```rust
  use serde::{Deserialize, Serialize};
  use specta::Type;

  pub mod manager;

  /// The canonical timeline event returned to the frontend.
  #[derive(Debug, Clone, Serialize, Deserialize, Type)]
  pub struct TimelineEvent {
      pub id: String,
      pub title: String,
      pub start_at: i64,       // Unix ms (UTC)
      pub end_at: i64,         // Unix ms (UTC)
      pub description: Option<String>,
      pub tags: Vec<String>,   // deserialized from JSON string in DB
      pub recording_id: Option<String>,
      pub document_id: Option<String>,
      pub created_at: i64,     // Unix ms (UTC)
  }

  /// Request body for creating a new event.
  #[derive(Debug, Clone, Serialize, Deserialize, Type)]
  pub struct CreateEventRequest {
      pub title: String,
      pub start_at: i64,
      pub end_at: i64,
      pub description: Option<String>,
      pub tags: Option<Vec<String>>,
      pub recording_id: Option<String>,
      pub document_id: Option<String>,
  }

  /// Request body for updating an existing event.
  /// All fields are optional; only `Some` fields are applied.
  #[derive(Debug, Clone, Serialize, Deserialize, Type)]
  pub struct UpdateEventRequest {
      pub title: Option<String>,
      pub start_at: Option<i64>,
      pub end_at: Option<i64>,
      pub description: Option<Option<String>>,   // Some(None) = clear the field
      pub tags: Option<Vec<String>>,
      pub recording_id: Option<Option<String>>,  // Some(None) = unlink
      pub document_id: Option<Option<String>>,   // Some(None) = unlink
  }

  /// Intermediate row type used only when reading from SQLite.
  /// `tags` arrives as a raw JSON string; `TimelineEvent::from(row)` deserializes it.
  #[derive(Debug)]
  pub(crate) struct TimelineEventRow {
      pub id: String,
      pub title: String,
      pub start_at: i64,
      pub end_at: i64,
      pub description: Option<String>,
      pub tags: Option<String>,         // raw JSON, e.g. `["meeting","client"]`
      pub recording_id: Option<String>,
      pub document_id: Option<String>,
      pub created_at: i64,
  }

  impl From<TimelineEventRow> for TimelineEvent {
      fn from(row: TimelineEventRow) -> Self {
          let tags = row
              .tags
              .as_deref()
              .and_then(|s| serde_json::from_str::<Vec<String>>(s).ok())
              .unwrap_or_default();
          Self {
              id: row.id,
              title: row.title,
              start_at: row.start_at,
              end_at: row.end_at,
              description: row.description,
              tags,
              recording_id: row.recording_id,
              document_id: row.document_id,
              created_at: row.created_at,
          }
      }
  }
  ```

  ```rust
  use sqlx::SqlitePool;
  use uuid::Uuid;

  use crate::error::AppError;
  use super::{CreateEventRequest, TimelineEvent, TimelineEventRow, UpdateEventRequest};

  pub struct TimelineManager {
      pool: SqlitePool,
  }

  impl TimelineManager {
      pub fn new(pool: SqlitePool) -> Self {
          Self { pool }
      }

      // ------------------------------------------------------------------ helpers

      /// Serialize `Vec<String>` to a compact JSON array string.
      fn tags_to_json(tags: &[String]) -> Result<String, AppError> {
          serde_json::to_string(tags)
              .map_err(|e| AppError::Storage(format!("tags serialize: {e}")))
      }

      /// Deserialize a JSON array string to `Vec<String>`.
      /// Returns an empty Vec on `None` or parse error (never errors out).
      fn json_to_tags(raw: Option<&str>) -> Vec<String> {
          raw.and_then(|s| serde_json::from_str(s).ok())
              .unwrap_or_default()
      }

      // ------------------------------------------------------------------ create

      pub async fn create_event(
          &self,
          req: CreateEventRequest,
      ) -> Result<TimelineEvent, AppError> {
          let id = Uuid::new_v4().to_string();
          let now = chrono::Utc::now().timestamp_millis();
          let tags_json = Self::tags_to_json(req.tags.as_deref().unwrap_or(&[]))?;

          sqlx::query(
              r#"
              INSERT INTO timeline_events
                  (id, title, start_at, end_at, description, tags, recording_id, document_id, created_at)
              VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
              "#,
          )
          .bind(&id)
          .bind(&req.title)
          .bind(req.start_at)
          .bind(req.end_at)
          .bind(&req.description)
          .bind(&tags_json)
          .bind(&req.recording_id)
          .bind(&req.document_id)
          .bind(now)
          .execute(&self.pool)
          .await
          .map_err(|e| AppError::Storage(e.to_string()))?;

          self.get_event_by_id(&id).await
      }

      // ------------------------------------------------------------------ read (single)

      async fn get_event_by_id(&self, id: &str) -> Result<TimelineEvent, AppError> {
          let row = sqlx::query_as!(
              TimelineEventRow,
              r#"
              SELECT id, title, start_at, end_at, description, tags,
                     recording_id, document_id, created_at
              FROM timeline_events
              WHERE id = ?
              "#,
              id
          )
          .fetch_optional(&self.pool)
          .await
          .map_err(|e| AppError::Storage(e.to_string()))?
          .ok_or_else(|| AppError::NotFound(format!("timeline event {id}")))?;

          Ok(TimelineEvent::from(row))
      }

      // ------------------------------------------------------------------ update

      pub async fn update_event(
          &self,
          id: &str,
          req: UpdateEventRequest,
      ) -> Result<TimelineEvent, AppError> {
          // Fetch current state first so we can apply partial updates.
          let current = self.get_event_by_id(id).await?;

          let title = req.title.unwrap_or(current.title);
          let start_at = req.start_at.unwrap_or(current.start_at);
          let end_at = req.end_at.unwrap_or(current.end_at);
          let description = req.description.unwrap_or(current.description);
          let tags = req.tags.unwrap_or(current.tags);
          let recording_id = req.recording_id.unwrap_or(current.recording_id);
          let document_id = req.document_id.unwrap_or(current.document_id);
          let tags_json = Self::tags_to_json(&tags)?;

          sqlx::query(
              r#"
              UPDATE timeline_events
              SET title = ?, start_at = ?, end_at = ?, description = ?,
                  tags = ?, recording_id = ?, document_id = ?
              WHERE id = ?
              "#,
          )
          .bind(&title)
          .bind(start_at)
          .bind(end_at)
          .bind(&description)
          .bind(&tags_json)
          .bind(&recording_id)
          .bind(&document_id)
          .bind(id)
          .execute(&self.pool)
          .await
          .map_err(|e| AppError::Storage(e.to_string()))?;

          self.get_event_by_id(id).await
      }

      // ------------------------------------------------------------------ delete

      pub async fn delete_event(&self, id: &str) -> Result<(), AppError> {
          let result = sqlx::query("DELETE FROM timeline_events WHERE id = ?")
              .bind(id)
              .execute(&self.pool)
              .await
              .map_err(|e| AppError::Storage(e.to_string()))?;

          if result.rows_affected() == 0 {
              return Err(AppError::NotFound(format!("timeline event {id}")));
          }
          Ok(())
      }

      // ------------------------------------------------------------------ list in range

      /// Returns all events whose `start_at` falls within [start_ms, end_ms] (inclusive),
      /// ordered by `start_at` ascending.
      pub async fn list_events_in_range(
          &self,
          start_ms: i64,
          end_ms: i64,
      ) -> Result<Vec<TimelineEvent>, AppError> {
          let rows = sqlx::query_as!(
              TimelineEventRow,
              r#"
              SELECT id, title, start_at, end_at, description, tags,
                     recording_id, document_id, created_at
              FROM timeline_events
              WHERE start_at >= ? AND start_at <= ?
              ORDER BY start_at ASC
              "#,
              start_ms,
              end_ms
          )
          .fetch_all(&self.pool)
          .await
          .map_err(|e| AppError::Storage(e.to_string()))?;

          Ok(rows.into_iter().map(TimelineEvent::from).collect())
      }

      // ------------------------------------------------------------------ search

      /// Case-insensitive LIKE search across `title` and `description`.
      pub async fn search_events(&self, query: &str) -> Result<Vec<TimelineEvent>, AppError> {
          let pattern = format!("%{}%", query.to_lowercase());
          let rows = sqlx::query_as!(
              TimelineEventRow,
              r#"
              SELECT id, title, start_at, end_at, description, tags,
                     recording_id, document_id, created_at
              FROM timeline_events
              WHERE LOWER(title) LIKE ? OR LOWER(description) LIKE ?
              ORDER BY start_at DESC
              "#,
              pattern,
              pattern
          )
          .fetch_all(&self.pool)
          .await
          .map_err(|e| AppError::Storage(e.to_string()))?;

          Ok(rows.into_iter().map(TimelineEvent::from).collect())
      }

      // ------------------------------------------------------------------ link helpers

      pub async fn link_recording(
          &self,
          event_id: &str,
          recording_id: &str,
      ) -> Result<(), AppError> {
          let result = sqlx::query(
              "UPDATE timeline_events SET recording_id = ? WHERE id = ?",
          )
          .bind(recording_id)
          .bind(event_id)
          .execute(&self.pool)
          .await
          .map_err(|e| AppError::Storage(e.to_string()))?;

          if result.rows_affected() == 0 {
              return Err(AppError::NotFound(format!("timeline event {event_id}")));
          }
          Ok(())
      }

      pub async fn link_document(
          &self,
          event_id: &str,
          document_id: &str,
      ) -> Result<(), AppError> {
          let result = sqlx::query(
              "UPDATE timeline_events SET document_id = ? WHERE id = ?",
          )
          .bind(document_id)
          .bind(event_id)
          .execute(&self.pool)
          .await
          .map_err(|e| AppError::Storage(e.to_string()))?;

          if result.rows_affected() == 0 {
              return Err(AppError::NotFound(format!("timeline event {event_id}")));
          }
          Ok(())
      }
  }
  ```

  ```typescript
  import {
    format,
    parseISO,
    startOfMonth,
    endOfMonth,
    startOfWeek,
    endOfWeek,
    startOfDay,
    endOfDay,
    eachDayOfInterval,

  /** Convert a Unix-ms timestamp (UTC) to a local-time Date object. */
  export function msToDate(ms: number): Date {
    return new Date(ms);
  }

  /** Convert a local Date object to a UTC Unix-ms integer. */
  export function dateToMs(date: Date): number {
    return date.getTime();
  }

  /**
   * Parse the string produced by an `<input type="datetime-local">` element
   * (format: "YYYY-MM-DDTHH:mm") to a UTC Unix-ms integer.
   *
   * The browser interprets datetime-local values in the user's local timezone,
   * so `new Date(value)` correctly yields the right UTC instant.
   */
  export function datetimeLocalToMs(value: string): number {
    return new Date(value).getTime();
  }

  /**
   * Format a UTC Unix-ms integer to the "YYYY-MM-DDTHH:mm" string expected by
   * `<input type="datetime-local">` (local timezone).
   */
  export function msToDatetimeLocal(ms: number): string {
    const d = new Date(ms);
    // Adjust for local timezone offset so the displayed value is local time.
    const offsetMs = d.getTimezoneOffset() * 60 * 1000;
    return new Date(d.getTime() - offsetMs).toISOString().slice(0, 16);
  }

  /** Format a UTC Unix-ms integer to a human-readable local time string. */
  export function formatEventTime(ms: number): string {
    return format(new Date(ms), "HH:mm");
  }

  /** Format a UTC Unix-ms integer to a full date-time label (e.g. "Mar 20, 14:30"). */
  export function formatEventDateTime(ms: number): string {
    return format(new Date(ms), "MMM d, HH:mm");
  }

  /** Format a UTC Unix-ms integer to a short date label (e.g. "Mar 20"). */
  export function formatEventDate(ms: number): string {
    return format(new Date(ms), "MMM d");
  }

  /** Return the start and end UTC-ms bounds for the month containing `date`. */
  export function monthBounds(date: Date): { start: number; end: number } {
    return {
      start: startOfMonth(date).getTime(),
      end: endOfMonth(date).getTime(),
    };
  }

  /** Return the start and end UTC-ms bounds for the ISO week containing `date`. */
  export function weekBounds(date: Date): { start: number; end: number } {
    return {
      start: startOfWeek(date, { weekStartsOn: 0 }).getTime(),
      end: endOfWeek(date, { weekStartsOn: 0 }).getTime(),
    };
  }

  /** Return the start and end UTC-ms bounds for the day containing `date`. */
  export function dayBounds(date: Date): { start: number; end: number } {
    return {
      start: startOfDay(date).getTime(),
      end: endOfDay(date).getTime(),
    };
  }

  /** Return an array of Date objects for every day in [startMs, endMs]. */
  export function daysInRange(startMs: number, endMs: number): Date[] {
    return eachDayOfInterval({ start: new Date(startMs), end: new Date(endMs) });
  }

  export { isSameDay, isSameMonth };
  ```

  ```typescript
  import {
    createTimelineEvent,
    updateTimelineEvent,
    deleteTimelineEvent,
    listTimelineEvents,
    searchTimelineEvents,
  } from "../lib/bindings";
  import type { TimelineEvent, CreateEventRequest, UpdateEventRequest } from "../lib/bindings";
  import { monthBounds } from "../lib/timeUtils";

  export type ViewMode = "month" | "week" | "day";

  interface TimelineState {
    // ---- data ----
    events: TimelineEvent[];
    viewRange: { start: Date; end: Date };
    selectedEventId: string | null;
    viewMode: ViewMode;
    isLoading: boolean;
    error: string | null;

    // ---- actions ----
    fetchRange: (start: Date, end: Date) => Promise<void>;
    search: (query: string) => Promise<void>;
    createEvent: (req: CreateEventRequest) => Promise<TimelineEvent>;
    updateEvent: (id: string, req: UpdateEventRequest) => Promise<TimelineEvent>;
    deleteEvent: (id: string) => Promise<void>;
    selectEvent: (id: string | null) => void;
    setViewMode: (mode: ViewMode) => void;
    setViewRange: (start: Date, end: Date) => void;
    navigatePrev: () => void;
    navigateNext: () => void;
    navigateToday: () => void;
  }

  export const useTimelineStore = create<TimelineState>((set, get) => {
    const today = new Date();
    const { start: ms0, end: ms1 } = monthBounds(today);

    return {
      events: [],
      viewRange: { start: new Date(ms0), end: new Date(ms1) },
      selectedEventId: null,
      viewMode: "month",
      isLoading: false,
      error: null,

      fetchRange: async (start, end) => {
        set({ isLoading: true, error: null });
        try {
          const events = await listTimelineEvents(
            start.getTime(),
            end.getTime()
          );
          set({ events, viewRange: { start, end }, isLoading: false });
        } catch (err) {
          set({ error: String(err), isLoading: false });
        }
      },

      search: async (query) => {
        if (!query.trim()) {
          // Reset to current view range when search is cleared.
          const { viewRange } = get();
          await get().fetchRange(viewRange.start, viewRange.end);
          return;
        }
        set({ isLoading: true, error: null });
        try {
          const events = await searchTimelineEvents(query);
          set({ events, isLoading: false });
        } catch (err) {
          set({ error: String(err), isLoading: false });
        }
      },

      createEvent: async (req) => {
        const event = await createTimelineEvent(req);
        set((s) => ({ events: [...s.events, event].sort((a, b) => a.start_at - b.start_at) }));
        return event;
      },

      updateEvent: async (id, req) => {
        const updated = await updateTimelineEvent(id, req);
        set((s) => ({
          events: s.events
            .map((e) => (e.id === id ? updated : e))
            .sort((a, b) => a.start_at - b.start_at),
        }));
        return updated;
      },

      deleteEvent: async (id) => {
        await deleteTimelineEvent(id);
        set((s) => ({
          events: s.events.filter((e) => e.id !== id),
          selectedEventId: s.selectedEventId === id ? null : s.selectedEventId,
        }));
      },

      selectEvent: (id) => set({ selectedEventId: id }),

      setViewMode: (mode) => set({ viewMode: mode }),

      setViewRange: (start, end) => {
        set({ viewRange: { start, end } });
        get().fetchRange(start, end);
      },

      navigatePrev: () => {
        const { viewRange, viewMode } = get();
        const ref = viewRange.start;
        let start: Date, end: Date;

        if (viewMode === "month") {
          start = new Date(ref.getFullYear(), ref.getMonth() - 1, 1);
          end = new Date(ref.getFullYear(), ref.getMonth(), 0, 23, 59, 59, 999);
        } else if (viewMode === "week") {
          start = new Date(ref.getTime() - 7 * 86_400_000);
          end = new Date(start.getTime() + 7 * 86_400_000 - 1);
        } else {
          start = new Date(ref.getTime() - 86_400_000);
          end = new Date(start.getTime() + 86_400_000 - 1);
        }
        get().setViewRange(start, end);
      },

      navigateNext: () => {
        const { viewRange, viewMode } = get();
        const ref = viewRange.start;
        let start: Date, end: Date;

        if (viewMode === "month") {
          start = new Date(ref.getFullYear(), ref.getMonth() + 1, 1);
          end = new Date(ref.getFullYear(), ref.getMonth() + 2, 0, 23, 59, 59, 999);
        } else if (viewMode === "week") {
          start = new Date(ref.getTime() + 7 * 86_400_000);
          end = new Date(start.getTime() + 7 * 86_400_000 - 1);
        } else {
          start = new Date(ref.getTime() + 86_400_000);
          end = new Date(start.getTime() + 86_400_000 - 1);
        }
        get().setViewRange(start, end);
      },

      navigateToday: () => {
        const today = new Date();
        const { viewMode } = get();
        let start: Date, end: Date;

        if (viewMode === "month") {
          start = new Date(today.getFullYear(), today.getMonth(), 1);
          end = new Date(today.getFullYear(), today.getMonth() + 1, 0, 23, 59, 59, 999);
        } else if (viewMode === "week") {
          const dow = today.getDay();
          start = new Date(today.getTime() - dow * 86_400_000);
          start.setHours(0, 0, 0, 0);
          end = new Date(start.getTime() + 7 * 86_400_000 - 1);
        } else {
          start = new Date(today);
          start.setHours(0, 0, 0, 0);
          end = new Date(start.getTime() + 86_400_000 - 1);
        }
        get().setViewRange(start, end);
      },
    };
  });
  ```
